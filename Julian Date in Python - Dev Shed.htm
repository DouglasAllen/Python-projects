Code:
>>> date = '2006-12-15'
>>> (year, month, day) = date.split('-')
>>> year = int(year)
>>> month = int(month)
>>> day = int(day)
>>> import time
>>> t = time.mktime((year, month, day, 0, 0, 0, 0, 0, 0))
>>> t
1166140800.0
>>> time.gmtime(t)
(2006, 12, 15, 0, 0, 0, 4, 349, 0)
>>> time.gmtime(t)[7]
349
>>>
